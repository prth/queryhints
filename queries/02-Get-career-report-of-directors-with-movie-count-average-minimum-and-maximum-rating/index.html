Get career report of directors with their movie count, average ratings, minimum rating, maximum rating

SQL

  • MySQL, PostgreSQL
select
  director,
  count(*),
  avg(rating),
  min(rating),
  max(rating)
from movies
group by director;

MongoDB

db.movies.aggregate([
  {
    $group: {
      _id: "$director",
      count: { $sum: 1 },
      avg: { $avg: "$rating" },
      min: { $min: "$rating" },
      max: { $max: "$rating" },
    },
  },
])
