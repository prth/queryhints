template: query
title: Get directors of top 10 rated movies
mongodb: |
  db.movies.aggregate([
    {
      $sort: { rating: -1 }
    },
    {
      $limit: 10
    },
    {
      $group: {
        _id: '$director'
      }
    }
  ]);
sql: |
  select distinct director 
  from movies
  order by rating desc
  limit 10;

#### SQL

- MySQL, PostgreSQL

```sql
select distinct director
from movies
order by rating desc
limit 10;
```
