title: Get hourly count of movie reviews
tags:
  - order
  - limit
dbs:
  - mysql
  - postgres
mysql: |
  select DATE_FORMAT(review_created_at, '%Y-%m-%d %H:00:00') as hour, count(*)
  FROM movie_reviews
  GROUP BY hour
  ORDER BY hour;
postgres: |
  select DATE_FORMAT(review_created_at, '%Y-%m-%d %H:00:00') as hour, count(*)
  FROM movie_reviews
  GROUP BY hour
  ORDER BY hour;
